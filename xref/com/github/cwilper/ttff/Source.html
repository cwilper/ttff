5   /**
6    * A sequence of objects.
7    * <p>
8    * This interface is modeled after {@link java.util.Iterator}, with the
9    * following differences:
10   * <ul>
11   *   <li> It is {@link Closeable}.</li>
12   *   <li> All methods (except <code>close()</code>) may throw an
13   *        {@link IOException}.</li>
14   *   <li> An additional method, {@link #peek()} allows access to the next
15   *        item without advancing the sequence.</li>
16   *   <li> There is no <code>remove()</code> method.</li>
17   * </ul>
18   *
19   * @param <T> the type over which the source operates.
20   * @see Sources
21   */
22  public interface Source<T> extends Closeable {
23  
24      /**
25       * Tells whether the source has any more objects.
26       *
27       * @return <code>true</code> if there are more objects, <code>false</code>
28       *         otherwise.
29       * @throws IOException if an I/O problem occurs.
30       */
31      boolean hasNext() throws IOException;
32  
33      /**
34       * Gets the next object, advancing the sequence by one.
35       *
36       * @return the next object.
37       * @throws IOException if an I/O problem occurs.
38       */
39      T next() throws IOException;
40  
41      /**
42       * Gets the next object without advancing the sequence.
43       *
44       * @return the next object.

47      T peek() throws IOException;
